<h2>Requirements</h2>
<ul>
    <li>composer</li>
    <li>PHP >= 7.1.3</li>
</ul>

<h2>Installation</h2>
<pre>composer install</pre>
<pre>php artisan migrate --seed</pre>

<h2>Local Deployment</h2>
<pre>php artisan serve</pre>
